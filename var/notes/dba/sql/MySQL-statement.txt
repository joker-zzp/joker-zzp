mysql 语句总结

  查看数据库
  show databases;

  返回当前数据库下的所有表的名称
  show tables;

  更换当前使用的数据库
  use databaseName;

  查看数据库编码格式：
  show variables like 'helloworld';

  查看数据表的编码格式:
  show create table <表名>;

  查看表结构:
  desc <表名>;
  describe <表名>;
  show columns from <表名>

  创建数据库时指定数据库的字符集:
  create database <数据库名> character set utf8;

  创建数据表时指定数据表的编码格式
  create table tb_books (
    name varchar(45) not null,
    price double not null,
    bookCount int not null,
    author varchar(45) not null ) default charset = utf8;

  更改数据库编码格式：
  alter database <数据库名> character set utf8;

  修改数据表格编码格式
  alter table <表名> character set utf8;

  修改字段编码格式
  alter table <表名> change <字段名> <字段名> <类型> character set utf8;
  alter table user change username username varchar(20) character set utf8 not null;

  查询用户的权限
  select * from mysql.user where user='admin' \G;

  创建表
  create table mysql_users(
    user_id int(6) primary key not null auto_increment comment '用户ID',
    user_name varchar(16) not null comment '用户名称',
    user_password varchar(32) not null comment '用户密码',
    user_data_base varchar(256) comment '数据库名称'
  ) comment '数据库的用户' character set utf8;

  插入数据
  INSERT INTO mysql_users (user_name,user_password,user_data_base)
                       VALUES
                       ('yuanfei','yuanfei','helloworld');
