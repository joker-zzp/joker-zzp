# ------------------------------
# hosts.allow 的优先级要高于 hosts.deny
#
#
# ------------------------------

# 编辑 /etc/hosts.allow 文件

ALL:218.64.87.0/255.255.255.128  <-- 从 218.64.87.0 - 128 可以访问
sshd:0.0.0.0:allow  <--设定ip可以远程登录
sshd:ALL  <--全部允许啥ssh远程登录


# 编辑 /etc/hosts.deny 文件

ALL:218.64.87.0/255.255.255.128  <-- 限制218.64.87.0 - 127 对所有服务的访问
sshd:ALL  <--全部拒绝啥ssh登录
sshd: 0.0.0.0/22  <--设定ip禁止远程访问

#
